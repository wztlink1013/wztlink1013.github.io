class Solution {
    public int findLengthOfLCIS(int[] nums) {
        if (nums.length == 0) { return 0; }
        int result = 1;
        int count = 1;
        for (int i=0; i&lt;nums.length-1; i++) {
            if (nums[i] &lt; nums[i+1] ) {
                count++;
                if (result &lt; count) {result = count;}
            } else {
                if (result &lt; count) {result = count;}
                count = 1;
            }
        }
        return result;
    }
}
