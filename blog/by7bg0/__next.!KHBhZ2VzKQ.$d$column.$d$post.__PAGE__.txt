    if(!root) return []
    let result = []
    let que = []
    que.push(root)
    while(que.length !== 0) {
        let init_size = que.length
        let temp = []
        for (let i = 0; i&lt;init_size; i++) {
            let mp = que.shift()
            if(mp.left) que.push(mp.left)
            if(mp.right) que.push(mp.right)
            temp.push(mp.val)
        }
        result.push(temp)
    }
