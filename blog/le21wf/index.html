    public void push(int x) {

        int temp = stack_input.size();
        if (stack_output.size() == 0) {
            for (int i=0; i<temp; i++){
                stack_output.push(stack_input.pop());
            }
        }

    public int pop() {
        int temp = stack_input.size();
        if (stack_output.size() == 0) {
            for (int i=0; i<temp; i++){
                stack_output.push(stack_input.pop());
            }
        }
        return stack_output.pop();
