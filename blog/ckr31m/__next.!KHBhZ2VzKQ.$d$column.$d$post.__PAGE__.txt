class Solution {
    public int calculate(String s) {
        Stack&lt;Integer&gt; stack_1 = new Stack&lt;Integer&gt;();
        int sum = 0;
        int sign = 1;
        int num = 0;

        for (int i=0;i&lt;s.length();i++) {
            char ch = s.charAt(i);
            if (Character.isDigit(ch)) {
                num = 10 * num + (int) (ch - '0');
            } else if (ch == '+') {
                num = num * sign;
                sum = sum + num;
                sign = 1;
                num = 0;
            } else if (ch == '-') {
                num = num * sign;
                sum = sum + num;
                sign = -1;
                num = 0;
            } else if (ch == '(') {
                stack_1.push(sum);
                stack_1.push(sign);
                sum = 0;
                num = 0;
                sign = 1;
            } else if (ch == ')') {
                num = num * sign;
                sum = sum + num;
                sum = stack_1.pop() * sum;
                sum = sum + stack_1.pop();
                num = 0;
                sign = 1;
            }

        }
        sum = sum + sign * num;
        return sum;
    }}
