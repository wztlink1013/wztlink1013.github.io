MatrixUDG::MatrixUDG()
{
    char c1, c2;
    int i, j, weight, p1, p2;
    
    // è¾“å…¥"é¡¶ç‚¹æ•°"å’Œ"è¾¹æ•°"
    cout << "input vertex number: ";
    cin >> mVexNum;
    cout << "input edge number: ";
    cin >> mEdgNum;
    if ( mVexNum < 1 || mEdgNum < 1 || (mEdgNum > (mVexNum * (mVexNum-1))))
    {
        cout << "input error: invalid parameters!" << endl;
        return ;
    }
    
    // åˆå§‹åŒ–"é¡¶ç‚¹"
    for (i = 0; i < mVexNum; i++)
    {
        cout << "vertex(" << i << "): ";
        mVexs[i] = readChar();
    }

    // 1. åˆå§‹åŒ–"è¾¹"çš„æƒå€¼
    for (i = 0; i < mVexNum; i++)
    {
        for (j = 0; j < mVexNum; j++)
        {
            if (i==j)
                mMatrix[i][j] = 0;
            else
                mMatrix[i][j] = INF;
        }
    }
    // 2. åˆå§‹åŒ–"è¾¹"çš„æƒå€¼: æ ¹æ®ç”¨æˆ·çš„è¾“å…¥è¿›è¡Œåˆå§‹åŒ–
    for (i = 0; i < mEdgNum; i++)
    {
        // è¯»å–è¾¹çš„èµ·å§‹é¡¶ç‚¹ï¼Œç»“æŸé¡¶ç‚¹ï¼Œæƒå€¼
        cout << "edge(" << i << "): ";
        c1 = readChar();
        c2 = readChar();
        cin >> weight;

        p1 = getPosition(c1);
        p2 = getPosition(c2);
        if (p1==-1 || p2==-1)
        {
            cout << "input error: invalid edge!" << endl;
            return ;
        }

        mMatrix[p1][p2] = weight;
        mMatrix[p2][p1] = weight;
    }
}

MatrixUDG::MatrixUDG(char vexs[], int vlen, int matrix[][9])
{
    int i, j;
    
    // åˆå§‹åŒ–"é¡¶ç‚¹æ•°"å’Œ"è¾¹æ•°"
    mVexNum = vlen;
    // åˆå§‹åŒ–"é¡¶ç‚¹"
    for (i = 0; i < mVexNum; i++)
        mVexs[i] = vexs[i];

    // åˆå§‹åŒ–"è¾¹"
    for (i = 0; i < mVexNum; i++)
        for (j = 0; j < mVexNum; j++)
            mMatrix[i][j] = matrix[i][j];

    // ç»Ÿè®¡è¾¹çš„æ•°ç›®
    for (i = 0; i < mVexNum; i++)
        for (j = 0; j < mVexNum; j++)
            if (i!=j && mMatrix[i][j]!=INF)
                mEdgNum++;
    mEdgNum /= 2;
}

/*
 * å¹¿åº¦ä¼˜å…ˆæœç´¢ï¼ˆç±»ä¼¼äºæ ‘çš„å±‚æ¬¡éå†ï¼‰
 */
void MatrixUDG::BFS()
{
    int head = 0;
    int rear = 0;
    int queue[MAX];     // è¾…ç»„é˜Ÿåˆ—
    int visited[MAX];   // é¡¶ç‚¹è®¿é—®æ ‡è®°
    int i, j, k;

    for (i = 0; i < mVexNum; i++)
        visited[i] = 0;

    cout << "BFS: ";
    for (i = 0; i < mVexNum; i++)
    {
        if (!visited[i])
        {
            visited[i] = 1;
            cout << mVexs[i] << " ";
            queue[rear++] = i;  // å…¥é˜Ÿåˆ—
        }
        while (head != rear) 
        {
            j = queue[head++];  // å‡ºé˜Ÿåˆ—
            for (k = firstVertex(j); k >= 0; k = nextVertex(j, k)) //kæ˜¯ä¸ºè®¿é—®çš„é‚»æ¥é¡¶ç‚¹
            {
                if (!visited[k])
                {
                    visited[k] = 1;
                    cout << mVexs[k] << " ";
                    queue[rear++] = k;
                }
            }
        }
    }
    cout << endl;
}

/*

void MatrixUDG::prim(int start)
{
    int min,i,j,k,m,n,sum;
    int index=0;         // primæœ€å°æ ‘çš„ç´¢å¼•ï¼Œå³primsæ•°ç»„çš„ç´¢å¼•
    char prims[MAX];     // primæœ€å°æ ‘çš„ç»“æœæ•°ç»„
    int weights[MAX];    // é¡¶ç‚¹é—´è¾¹çš„æƒå€¼

    // primæœ€å°ç”Ÿæˆæ ‘ä¸­ç¬¬ä¸€ä¸ªæ•°æ˜¯"å›¾ä¸­ç¬¬startä¸ªé¡¶ç‚¹"ï¼Œå› ä¸ºæ˜¯ä»startå¼€å§‹çš„ã€‚
    prims[index++] = mVexs[start];

    // åˆå§‹åŒ–"é¡¶ç‚¹çš„æƒå€¼æ•°ç»„"ï¼Œ
    // å°†æ¯ä¸ªé¡¶ç‚¹çš„æƒå€¼åˆå§‹åŒ–ä¸º"ç¬¬startä¸ªé¡¶ç‚¹"åˆ°"è¯¥é¡¶ç‚¹"çš„æƒå€¼ã€‚
    for (i = 0; i < mVexNum; i++ )
        weights[i] = mMatrix[start][i];
    // å°†ç¬¬startä¸ªé¡¶ç‚¹çš„æƒå€¼åˆå§‹åŒ–ä¸º0ã€‚
    // å¯ä»¥ç†è§£ä¸º"ç¬¬startä¸ªé¡¶ç‚¹åˆ°å®ƒè‡ªèº«çš„è·ç¦»ä¸º0"ã€‚
    weights[start] = 0;

    for (i = 0; i < mVexNum; i++)
    {
        // ç”±äºä»startå¼€å§‹çš„ï¼Œå› æ­¤ä¸éœ€è¦å†å¯¹ç¬¬startä¸ªé¡¶ç‚¹è¿›è¡Œå¤„ç†ã€‚
        if(start == i)
            continue;

        j = 0;
        k = 0;
        min = INF;
        // åœ¨æœªè¢«åŠ å…¥åˆ°æœ€å°ç”Ÿæˆæ ‘çš„é¡¶ç‚¹ä¸­ï¼Œæ‰¾å‡ºæƒå€¼æœ€å°çš„é¡¶ç‚¹ã€‚
        while (j < mVexNum)
        {
            // è‹¥weights[j]=0ï¼Œæ„å‘³ç€"ç¬¬jä¸ªèŠ‚ç‚¹å·²ç»è¢«æ’åºè¿‡"(æˆ–è€…è¯´å·²ç»åŠ å…¥äº†æœ€å°ç”Ÿæˆæ ‘ä¸­)ã€‚
            if (weights[j] != 0 && weights[j] < min)
            {
                min = weights[j];
                k = j;
            }
            j++;
        }

        // ç»è¿‡ä¸Šé¢çš„å¤„ç†åï¼Œåœ¨æœªè¢«åŠ å…¥åˆ°æœ€å°ç”Ÿæˆæ ‘çš„é¡¶ç‚¹ä¸­ï¼Œæƒå€¼æœ€å°çš„é¡¶ç‚¹æ˜¯ç¬¬kä¸ªé¡¶ç‚¹ã€‚
        // å°†ç¬¬kä¸ªé¡¶ç‚¹åŠ å…¥åˆ°æœ€å°ç”Ÿæˆæ ‘çš„ç»“æœæ•°ç»„ä¸­
        prims[index++] = mVexs[k];
        // å°†"ç¬¬kä¸ªé¡¶ç‚¹çš„æƒå€¼"æ ‡è®°ä¸º0ï¼Œæ„å‘³ç€ç¬¬kä¸ªé¡¶ç‚¹å·²ç»æ’åºè¿‡äº†(æˆ–è€…è¯´å·²ç»åŠ å…¥äº†æœ€å°æ ‘ç»“æœä¸­)ã€‚
        weights[k] = 0;
        // å½“ç¬¬kä¸ªé¡¶ç‚¹è¢«åŠ å…¥åˆ°æœ€å°ç”Ÿæˆæ ‘çš„ç»“æœæ•°ç»„ä¸­ä¹‹åï¼Œæ›´æ–°å…¶å®ƒé¡¶ç‚¹çš„æƒå€¼ã€‚
        for (j = 0 ; j < mVexNum; j++)
        {
            // å½“ç¬¬jä¸ªèŠ‚ç‚¹æ²¡æœ‰è¢«å¤„ç†ï¼Œå¹¶ä¸”éœ€è¦æ›´æ–°æ—¶æ‰è¢«æ›´æ–°ã€‚
            if (weights[j] != 0 && mMatrix[k][j] < weights[j])
                weights[j] = mMatrix[k][j];
        }
    }

    // è®¡ç®—æœ€å°ç”Ÿæˆæ ‘çš„æƒå€¼
    sum = 0;
    for (i = 1; i < index; i++)
    {
        min = INF;
        // è·å–prims[i]åœ¨mMatrixä¸­çš„ä½ç½®
        n = getPosition(prims[i]);
        // åœ¨vexs[0...i]ä¸­ï¼Œæ‰¾å‡ºåˆ°jçš„æƒå€¼æœ€å°çš„é¡¶ç‚¹ã€‚
        for (j = 0; j < i; j++)
        {
            m = getPosition(prims[j]);
            if (mMatrix[m][n]<min)
                min = mMatrix[m][n];
        }
        sum += min;
    }
    // æ‰“å°æœ€å°ç”Ÿæˆæ ‘
    cout << "PRIM(" << mVexs[start] << ")=" << sum << ": ";
    for (i = 0; i < index; i++)
        cout << prims[i] << " ";
    cout << endl;
}

/* 
 * è·å–å›¾ä¸­çš„è¾¹

EData* MatrixUDG::getEdges()
{
    int i,j;
    int index=0;
    EData *edges;

    edges = new EData[mEdgNum];
    for (i=0; i < mVexNum; i++)
    {
        for (j=i+1; j < mVexNum; j++)
        {
            if (mMatrix[i][j]!=INF)
            {
                edges[index].start  = mVexs[i];
                edges[index].end    = mVexs[j];
                edges[index].weight = mMatrix[i][j];
                index++;
            }
        }
    }

    return edges;
}

/* 
 * å¯¹è¾¹æŒ‰ç…§æƒå€¼å¤§å°è¿›è¡Œæ’åº(ç”±å°åˆ°å¤§)
 */
void MatrixUDG::sortEdges(EData* edges, int elen)
{
    int i,j;

    for (i=0; i<elen; i++)
    {
        for (j=i+1; j<elen; j++)
        {
            if (edges[i].weight > edges[j].weight)
            {
                // äº¤æ¢"è¾¹i"å’Œ"è¾¹j"
                swap(edges[i], edges[j]);
            }
        }
    }
}

{
    int i,m,n,p1,p2;
    int length;

    int vends[MAX]={0};     // ç”¨äºä¿å­˜"å·²æœ‰æœ€å°ç”Ÿæˆæ ‘"ä¸­æ¯ä¸ªé¡¶ç‚¹åœ¨è¯¥æœ€å°æ ‘ä¸­çš„ç»ˆç‚¹ã€‚

    {

        p2 = getPosition(edges[i].end);        // è·å–ç¬¬iæ¡è¾¹çš„"ç»ˆç‚¹"çš„åºå·

        m = getEnd(vends, p1);                 // è·å–p1åœ¨"å·²æœ‰çš„æœ€å°ç”Ÿæˆæ ‘"ä¸­çš„ç»ˆç‚¹

        // å¦‚æœm!=nï¼Œæ„å‘³ç€"è¾¹i"ä¸"å·²ç»æ·»åŠ åˆ°æœ€å°ç”Ÿæˆæ ‘ä¸­çš„é¡¶ç‚¹"æ²¡æœ‰å½¢æˆç¯è·¯
        if (m != n)

            rets[index++] = edges[i];           // ä¿å­˜ç»“æœ

    delete[] edges;

    // ç»Ÿè®¡å¹¶æ‰“å°"kruskalæœ€å°ç”Ÿæˆæ ‘"çš„ä¿¡æ¯
    length = 0;

void MatrixUDG::dijkstra(int vs, int prev[], int dist[])
{
    int i,j,k;
    int min;
    int tmp;
    int flag[MAX];      // flag[i]=1è¡¨ç¤º"é¡¶ç‚¹vs"åˆ°"é¡¶ç‚¹i"çš„æœ€çŸ­è·¯å¾„å·²æˆåŠŸè·å–ã€‚
    
    // åˆå§‹åŒ–
    for (i = 0; i < mVexNum; i++)
    {
        flag[i] = 0;              // é¡¶ç‚¹içš„æœ€çŸ­è·¯å¾„è¿˜æ²¡è·å–åˆ°ã€‚
        prev[i] = 0;              // é¡¶ç‚¹içš„å‰é©±é¡¶ç‚¹ä¸º0ã€‚
        dist[i] = mMatrix[vs][i]; // é¡¶ç‚¹içš„æœ€çŸ­è·¯å¾„ä¸º"é¡¶ç‚¹vs"åˆ°"é¡¶ç‚¹i"çš„æƒã€‚
    }

    // å¯¹"é¡¶ç‚¹vs"è‡ªèº«è¿›è¡Œåˆå§‹åŒ–
    flag[vs] = 1;
    dist[vs] = 0;

    // éå†mVexNum-1æ¬¡ï¼›æ¯æ¬¡æ‰¾å‡ºä¸€ä¸ªé¡¶ç‚¹çš„æœ€çŸ­è·¯å¾„ã€‚
    for (i = 1; i < mVexNum; i++)
    {
        // å¯»æ‰¾å½“å‰æœ€å°çš„è·¯å¾„ï¼›
        // å³ï¼Œåœ¨æœªè·å–æœ€çŸ­è·¯å¾„çš„é¡¶ç‚¹ä¸­ï¼Œæ‰¾åˆ°ç¦»vsæœ€è¿‘çš„é¡¶ç‚¹(k)ã€‚
        min = INF;
        for (j = 0; j < mVexNum; j++)
        {
            if (flag[j]==0 && dist[j]<min)
            {
                min = dist[j];
                k = j;
            }
        }
        // æ ‡è®°"é¡¶ç‚¹k"ä¸ºå·²ç»è·å–åˆ°æœ€çŸ­è·¯å¾„
        flag[k] = 1;

        // ä¿®æ­£å½“å‰æœ€çŸ­è·¯å¾„å’Œå‰é©±é¡¶ç‚¹
        // å³ï¼Œå½“å·²ç»"é¡¶ç‚¹kçš„æœ€çŸ­è·¯å¾„"ä¹‹åï¼Œæ›´æ–°"æœªè·å–æœ€çŸ­è·¯å¾„çš„é¡¶ç‚¹çš„æœ€çŸ­è·¯å¾„å’Œå‰é©±é¡¶ç‚¹"ã€‚
        for (j = 0; j < mVexNum; j++)
        {
            tmp = (mMatrix[k][j]==INF ? INF : (min + mMatrix[k][j]));
            if (flag[j] == 0 && (tmp  < dist[j]) )
            {
                dist[j] = tmp;
                prev[j] = k;
            }
        }
    }

    // æ‰“å°dijkstraæœ€çŸ­è·¯å¾„çš„ç»“æœ
    cout << "dijkstra(" << mVexs[vs] << "): " << endl;
    for (i = 0; i < mVexNum; i++)
        cout << "  shortest(" << mVexs[vs] << ", " << mVexs[i] << ")=" << dist[i] << endl;
}
